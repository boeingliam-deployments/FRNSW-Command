let store = JSON.parse(localStorage.getItem('frnsw_v3') || '{"activeId": null, "incidents": []}'); const STATUSES = ['responding', 'scene', 'returning', 'available']; // HELPERS function save() { localStorage.setItem('frnsw_v3', JSON.stringify(store)); } function getActive() { return store.incidents.find(i => i.id === store.activeId); } // QRF & NOTEPAD (Draggable logic remains same) function toggleQRF() { const hud = document.getElementById('qrfHud'); hud.style.display = (hud.style.display === 'flex') ? 'none' : 'flex'; } function toggleNotepad() { const hud = document.getElementById('notepadHud'); hud.style.display = (hud.style.display === 'flex') ? 'none' : 'flex'; } makeDraggable(document.getElementById('qrfHud'), document.getElementById('qrfHeader')); makeDraggable(document.getElementById('notepadHud'), document.getElementById('notepadHeader')); document.getElementById('notepadText').value = localStorage.getItem('frnsw_notepad') || ''; function makeDraggable(el, handle) { let dragging = false, ox = 0, oy = 0; handle.addEventListener('mousedown', e => { dragging = true; ox = el.offsetLeft - e.clientX; oy = el.offsetTop - e.clientY; }); document.addEventListener('mouseup', () => dragging = false); document.addEventListener('mousemove', e => { if (dragging) { el.style.left = (e.clientX + ox) + 'px'; el.style.top = (e.clientY + oy) + 'px'; } }); } // INCIDENT CORE LOGIC function createNewIncident() { const name = prompt("Enter Incident Identifier (e.g. P25-12345):"); if (!name) return; const newInc = { id: Date.now(), name: name.toUpperCase(), units: [], logs: [{ts: new Date().toLocaleTimeString(), msg: `Incident ${name.toUpperCase()} Created`}], status: 'active' }; store.incidents.push(newInc); store.activeId = newInc.id; save(); render(); } function switchIncident(id) { store.activeId = id; save(); render(); } function deleteIncidentById(id) { const inc = store.incidents.find(i => i.id === id); if (!inc) return; if (!confirm(`Delete incident ${inc.name}?`)) return; store.incidents = store.incidents.filter(i => i.id !== id); if (store.activeId === id) { store.activeId = store.incidents.length ? store.incidents[0].id : null; } save(); render(); } function openEditModal() { const inc = getActive(); if (!inc) return alert("No active incident to manage."); document.getElementById('editName').value = inc.name; document.getElementById('editStatus').value = inc.status || 'active'; document.getElementById('editModal').style.display = 'flex'; } function closeEditModal() { document.getElementById('editModal').style.display = 'none'; } function saveEdit() { const inc = getActive(); const newName = document.getElementById('editName').value.toUpperCase(); const newStatus = document.getElementById('editStatus').value; if (!newName) return; inc.name = newName; inc.status = newStatus; log(`Update: Name to ${newName}, Status: ${newStatus.toUpperCase()}`); save(); render(); closeEditModal(); } // UNIT LOGIC function addUnit() { const inc = getActive(); if (!inc || inc.status === 'suspended') return; const cs = document.getElementById('inCS').value.toUpperCase(); const type = document.getElementById('inType').value || 'Pumper'; if (!cs) return; const unit = { id: Date.now(), cs: cs, type: type, status: 'responding', location: '', isCmd: document.getElementById('inIsCmd').value === 'true', ts: new Date().toLocaleTimeString() }; inc.units.push(unit); log(`${unit.cs} assigned to board`); save(); render(); document.getElementById('inCS').value = ''; document.getElementById('inCS').focus(); } function setStatus(unitId, status) { const inc = getActive(); if (inc.status === 'suspended') return; const unit = inc.units.find(u => u.id === unitId); const oldStatus = unit.status; unit.status = status; unit.ts = new Date().toLocaleTimeString(); log(`${unit.cs} status changed from ${oldStatus.toUpperCase()} to ${status.toUpperCase()}`); save(); render(); } function updateLocation(unitId, val) { const inc = getActive(); const unit = inc.units.find(u => u.id === unitId); if(unit) { unit.location = val; save(); // Critical: save on every change } } function toggleCommand(unitId) { const inc = getActive(); if (inc.status === 'suspended') return; const unit = inc.units.find(u => u.id === unitId); unit.isCmd = !unit.isCmd; log(`${unit.cs} command role toggled ${unit.isCmd ? 'ON' : 'OFF'}`); save(); render(); } function removeUnit(unitId) { const inc = getActive(); const unit = inc.units.find(u => u.id === unitId); if(!confirm(`Remove ${unit.cs}?`)) return; inc.units = inc.units.filter(u => u.id !== unitId); log(`${unit.cs} released`); save(); render(); } function log(msg) { const inc = getActive(); if (!inc) return; inc.logs.unshift({ ts: new Date().toLocaleTimeString(), msg }); } // UI RENDER function render() { const inc = getActive(); const strip = document.getElementById('incTabStrip'); const main = document.getElementById('main-content'); const toolbar = document.getElementById('toolbar'); const unitTable = document.getElementById('unitTable'); // Render Tabs if (!store.incidents.length) { strip.innerHTML = '
No incidents.
'; unitTable.innerHTML = 'Ready for dispatch.'; ['responding','scene','returning','total'].forEach(s => document.getElementById('stat-'+s).textContent = '0'); document.getElementById('log').innerHTML = ''; return; } strip.innerHTML = store.incidents.map(i => { const isActive = i.id === store.activeId; return `
${i.name}
${i.status.toUpperCase()}
`; }).join(''); if (!inc) return; // Update Stats document.getElementById('stat-total').textContent = inc.units.length; ['responding', 'scene', 'returning'].forEach(s => { document.getElementById('stat-'+s).textContent = inc.units.filter(u => u.status === s).length; }); // Handle Suspended State UI const isSuspended = inc.status === 'suspended'; toolbar.style.opacity = isSuspended ? '0.2' : '1'; toolbar.style.pointerEvents = isSuspended ? 'none' : 'all'; const oldOverlay = main.querySelector('.suspended-overlay'); if (oldOverlay) oldOverlay.remove(); if (isSuspended) { const overlay = document.createElement('div'); overlay.className = 'suspended-overlay'; overlay.innerHTML = '
Operations Paused
'; main.appendChild(overlay); } // Render Units unitTable.innerHTML = inc.units.sort((a,b) => b.isCmd - a.isCmd).map(u => `
${u.isCmd ? 'IC ' : ''}${u.cs}
${u.type}
${STATUSES.map(s => ``).join('')}
`).join(''); document.getElementById('log').innerHTML = inc.logs.map(l => `
[${l.ts}] ${l.msg}
`).join(''); } function init() { render(); } setInterval(() => { document.getElementById('clock').textContent = new Date().toLocaleTimeString(); }, 1000); init();
